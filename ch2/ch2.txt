def makeLong(dfi, mmv, proc):
    aa = pd.concat([times,dfi], axis=1)
    bb = pd.melt(aa, id_vars=['julian','year','month','day'],
        var_name='id', value_name='z')
    mask = bb.z <= mmv
    cc = bb[~mask]
    cc['proc'] = proc
    return(cc)
